% Use a sourcemap to insert LaTeX markup into the author field for the
% specific author we want to highlight. This replaces occurrences of
% "Papazov, Hristo" with "\textbf{Papazov, Hristo}" so the family name
% is bold in the final bibliography. This relies on biber processing the
% .bib and must be defined before \addbibresource (so the mapping is in
% effect when the .bib is read).
\DeclareSourcemap{
  \maps[datatype=bibtex]{
    \map{
      \step[fieldsource=author,
            match=\regexp{Papazov,\s*Hristo},
            replace={\\textbf{H. Papazov}}]
    }
  }
}
